28// Shape object, responsible for generating and storing 2D Shapes.
29// The data here is used for both Rendering and Physics.
30// This Shape structure is made to be malliable, and allow for a lot of freedom.
31// Do note that having a lot of verticies is performance intensive, and its not suggested to run isConvex too much
32// if possible know if the shape is convex or not, or just simplify the collisions to always be or not be convex
33//
34// A dynamic array of Verticies. Normalized between [0, 1], (0, 0) being top left.
35// This way the size of your Object doesn't matter, and can be transposed to whatever Size freely.
36//
37// @var normalized_vertices defines the shape. With a literal line being drawn from each point to the next.
38// The vertices don't need to be sorted in any specific order, since its however you want your shape to be defined as.
39// The last vertex in the vector of vertices will automatically be drawn to the [0]th index vertex.
40//
41// Shapes need to be deallocated.
42//
43// Layout: [Vector, Pose, uint32, bool]
44// Size: 85(88), 53(56)
45typedef struct Shape {
46 struct VertexVector normalized_vertices; // Dynamic Array of Normalized Verticies. This defines the shape.
47 struct PoseVector vertex_poses; // Dynamic Array of Vertex Positions. This is the vertices position in world. Don't manipulate
48 struct Pose size; // Size of Shape, used to Transpose the Normalized Verticies.
49 uint32_t color; // *Until Textures become a thing.
50 bool convex; // Whether a Shape is Convex or Concave.
51} Shape;
